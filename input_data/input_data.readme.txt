# Your input data goes in this folder

format = .fastq.gz

Note: Currently, the workflow is written with paired-end NextSeq500 input data
in mind. If you received multiple files for each sample from your facility then
they should be concatenated before running.

ex. 

JWG3_1_R1.fastq.gz
JWG3_1_R2.fastq.gz
JWG3_2_R1.fastq.gz
JWG3_2_R2.fastq.gz
